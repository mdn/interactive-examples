const person = {
  isHuman: true,
  sayName: function () {
    console.log(`My name is ${this.name}`);
  }
};

const me = Object.create(person);
me.name = "Matthew"; // name is now a property set on the me object, but not on the person object

me.isHuman == true;
me.sayName() // -> "My name is Matthew"
