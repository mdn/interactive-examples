var startDate = new Date(Date.UTC(2007, 0, 10, 10, 0, 0)); // > 'Wed, 10 Jan 2007 10:00:00 GMT'
var endDate = new Date(Date.UTC(2007, 0, 10, 11, 0, 0));   // > 'Wed, 10 Jan 2007 11:00:00 GMT'

var dateTimeFormat = new Intl.DateTimeFormat("en", {
    hour: 'numeric',
    minute: 'numeric'
});

var parts = dateTimeFormat.formatRangeToParts(startDate, endDate);
for (let part of parts) {
  console.log(part);
}
