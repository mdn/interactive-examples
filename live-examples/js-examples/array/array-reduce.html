console.log("idx  acc  cur  arr");

const reducer = (acc, cur, idx, src) => {
    console.log(` ${idx}    ${acc}    ${cur}   [${src.toString()}]`);
    return acc + cur;
}

var result = [3,1,6].reduce(reducer, 0)

console.log("result: " + result );

// expected output
// 
// idx  acc  cur  arr
//  0    0    3   [3,1,6]
//  1    3    1   [3,1,6]
//  2    4    6   [3,1,6]
// final: 10
