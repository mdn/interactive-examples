function firstAlphabetical(locale, letter1, letter2) {
  if (new Intl.Collator(locale).compare(letter1, letter2) > 0) {
    return letter1;
  }
  return letter2;
}

console.log(firstAlphabetical('de', 'z', 'ä'));
// expected output: "z"

console.log(firstAlphabetical('sv', 'z', 'ä'));
// expected output: "ä"
