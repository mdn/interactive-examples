var paragraph = 'The quick brown fox jumped over the lazy dog. If the dog barked, was it really lazy?';

var searchTerm = 'dog';

console.log('The index of the first "' + searchTerm + '" from the beginning is ' + paragraph.indexOf(searchTerm));
// expected output: "The index of the first "dog" from the beginning is 41"

// Add a fromIndex
var skipFirst = paragraph.indexOf(searchTerm) + 1;

console.log('The index of the 2nd "' + searchTerm + '" is ' + paragraph.indexOf(searchTerm, skipFirst));
// expected output: "The index of the 2nd "dog" is 53"
